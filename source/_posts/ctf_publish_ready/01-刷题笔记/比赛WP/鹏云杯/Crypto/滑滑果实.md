# 滑滑果实

# 题目

```python
from Crypto.Util.number import *
from random import choice
from secret import flag

flag=flag.decode('utf-8')+ "1" * 30
def getMyPrime(nbits):
    while True:
        p = 1
        while p.bit_length() <= nbits:
            p *= choice(sieve_base)
      
        if isPrime(p-1):
            return p-1

p = getMyPrime(256)
q = getMyPrime(256)

n = p*q
e = 65537
m = bytes_to_long(flag.encode('utf-8'))

c = pow(m, e, n)

print(f'n = {n}')
print(f'e = {e}')
print(f'c = {c}')
```
